import numpy as np
import matplotlib.pyplot as plt
from scipy.signal import firwin, lfilter
from scipy.fft import fft

fs = 5000 # Sampling rate (Hz)
n_samples = 1024 # N = Number of samples
fs_nyq = fs / 2 # Nyquist frequency
notch_width = 50  # Notch filter bandwidth (Hz

# Generate a random frequency below the Nyquist frequency
np.random.seed(1)
rand_freq = np.random.uniform(0, fs_nyq) # Randomly chosen frequency for the sin signal (below the Nyquist frequency) 
t = np.arange(n_samples) / fs  # Create a time vector
sin_sig = np.sin(2 * np.pi * rand_freq * t)  # Generate a sin wave signal 

# Set the filter target frequency to the random frequency
f_filter = rand_freq 

print(rand_freq)

# plot the sin wave at 200 sampling points we generate
start_idx, end_idx = 0, 200 #only 200 sampling points
plt.figure(figsize=(12,8))
plt.plot(t[start_idx:end_idx], sin_sig[start_idx:end_idx], color='blue', linewidth=1)
plt.title(f"sin signal {rand_freq:.2f} Hz")
plt.xlabel("Time")
plt.ylabel("Amplitude")
plt.grid()
plt.show()

filter_lengths = [5, 11, 51, 101, 251 , 1001]  # Different filter lengths (only odd values to ensure symmetry)
attenuations = []   # # List to store signal amplitudes based on FFT

# Grid settings
n_filters = 6
n_cols = 3 
n_rows = 2  
fig, axes = plt.subplots(n_rows, n_cols, figsize=(15, 10))
axes = axes.flatten()

for i,filter_order in enumerate(filter_lengths):
    # Create FIR Notch filter
    fir_notch = firwin(filter_order, [rand_freq - notch_width / 2, rand_freq + notch_width / 2],pass_zero="bandstop", fs=fs)
    
    # Apply the filter to the signal
    filtered_signal = lfilter(fir_notch, 1.0, sin_sig)
    
    # Calculate FFT
    fft_original = np.abs(np.fft.fft(sin_sig)[:n_samples // 2]) / (n_samples / 2)
    fft_filtered = np.abs(np.fft.fft(filtered_signal)[:n_samples // 2]) / (n_samples / 2)
    freqs = np.fft.fftfreq(n_samples, 1 / fs)[:n_samples // 2]   
    
    # Find amplitude at the target frequency
    idx = np.argmin(np.abs(freqs - f_filter))
    attenuations.append(fft_filtered[idx])  # Amplitude store

    print(f"Filter length: {filter_order},attenuation at {f_filter} Hz, Amplitude: {fft_filtered[idx]:.4f}")
    
    # Visualize the effect of filter length - spectrum graph for each filter
    axes[i].plot(freqs, fft_filtered, label="Filtered Signal",linestyle="--", color='orange', linewidth=4)
    axes[i].plot(freqs, fft_original, label="Original Signal")
    axes[i].set_title(f"Filter Length {filter_order}")
    axes[i].set_xlabel("Frequency (Hz)")
    axes[i].set_ylabel("Amplitude")
    axes[i].legend()
    axes[i].grid()

plt.tight_layout()
plt.show()
